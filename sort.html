📊 Basic Sorting - Beginner

📌 Remember:
  • .sort() with strings: just call it!
  • .sort() with numbers: use (a, b) => a - b for ascending
  • (a, b) => b - a for descending
  • For objects: compare properties like a.age - b.age

1. Sort Strings Alphabetically

let words = ['dog', 'cat', 'elephant', 'bird', 'ant'];
words.sort();
console.log(words);

2. Sort Numbers Ascending

let numbers = [42, 3, 15, 8, 23, 1];
numbers.sort((a, b) => a - b);
console.log(numbers);

3. Sort Numbers Descending

let scores = [85, 92, 78, 95, 88, 76];
scores.sort((a, b) => b - a);
console.log(scores);

4. Sort by String Length

let animals = ['elephant', 'cat', 'butterfly', 'dog', 'ant'];
animals.sort((a, b) => a.length - b.length);
console.log(animals);

5. Sort Objects by Age

let people = [
    { name: 'Alice', age: 30 },
    { name: 'Bob', age: 25 },
    { name: 'Charlie', age: 35 },
    { name: 'Diana', age: 28 }
];
people.sort((a, b) => a.age - b.age);
console.log(people);

🎁 Bonus: Sort by Grade then Name

let students = [
    { name: 'Alex', grade: 85 },
    { name: 'Bella', grade: 92 },
    { name: 'Charlie', grade: 85 },
    { name: 'Diana', grade: 92 },
    { name: 'Evan', grade: 78 }
];

students.sort((a, b) => {
    if (a.grade !== b.grade) {
        return b.grade - a.grade; // Higher grade first
    } else {
        return a.name.localeCompare(b.name); // Alphabetical if same grade
    }
});
console.log(students);
